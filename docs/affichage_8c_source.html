147  } else {
148  sprintf(buffer, "%s %s", statLabel, multPart);

536 void creerDescr(InventoryUI *ui, t_item *item, SDL_Renderer *renderer, t_input *input) {
537  char processedDesc[256];
538  strcpy(processedDesc, item->description);
539 
540  // Remplacer les "\\n" par "\n"
541  char *found;
542  while ((found = strstr(processedDesc, "\\n")) != NULL) {
543  *found = '\n'; // Remplacer le premier '\' par '\n'
544  memmove(found + 1, found + 2, strlen(found + 2) + 1); // Décaler le reste
545  }
546 
547  int i = 0, j, nb = 1;
548  while (processedDesc[i++] != '\0') {
549  for (j = 0; processedDesc[i] != '\n' && processedDesc[i] != '\0'; i++, j++) {
550  processedDesc[j] = processedDesc[i];
551  }
552  processedDesc[j] = '\0';
553 
554  ui->ecrit->description[nb - 1] = createText(renderer, processedDesc, ui->ext->descr_font, ui->ecrit->color_txt);
555  ui->ecrit->description[nb - 1]->rect.x = ui->elems->descrItem.rect.x + input->windowWidth * 0.005;
556  ui->ecrit->description[nb - 1]->rect.y = ui->ecrit->text_descr->rect.y + input->windowHeight * 0.03 * nb;
557  nb++;
558  ui->ecrit->descr[0] = '\0';
559  }
560  ui->ecrit->count_descr = nb - 1;
561 
562  ui->ecrit->nom_txt_item[0] = createStatLine("Health Max : ", item->stats.healthMax.additive, item->stats.healthMax.multiplicative);
563  ui->ecrit->nom_txt_item[1] = createStatLine("Attack : ", item->stats.attack.additive, item->stats.attack.multiplicative);
564  ui->ecrit->nom_txt_item[2] = createStatLine("Defense : ", item->stats.defense.additive, item->stats.defense.multiplicative);
565  ui->ecrit->nom_txt_item[3] = createStatLine("Speed : ", item->stats.speed.additive, item->stats.speed.multiplicative);
566 
567  for (int i = 0; i < 4; i++) {
568  ui->ecrit->text_item[i] = createText(renderer, ui->ecrit->nom_txt_item[i], ui->ext->descr_font, ui->ecrit->color_txt);
569  ui->ecrit->text_player[i] = createText(renderer, ui->ecrit->nom_txt_player[i], ui->ext->item_font, ui->ecrit->color_txt);
570  }
571 
572  ui->ecrit->text_player[0]->rect.x = ui->elems->statsPlayer.rect.x + input->windowWidth * 0.01;
573  ui->ecrit->text_player[0]->rect.y = ui->elems->statsPlayer.rect.y + input->windowHeight * 0.01;
574  ui->ecrit->text_item[0]->rect.x = ui->elems->statsItem.rect.x + input->windowWidth * 0.01;
575  ui->ecrit->text_item[0]->rect.y = ui->elems->statsItem.rect.y + input->windowHeight * 0.01;
576 
577  ui->itemclique = item;
578  ui->peutEquiper = 1;
579 }

602 void freeInv(void *elt) {
603  InventoryUI *ui = (InventoryUI *)elt;
604  if (!ui) return;
605 
606  // Libération de la structure externe
607  if (ui->ext) {
608  // Ne pas libérer character ici car il est géré ailleurs
609  /* Liberation des polices */
610  if (ui->ext->item_font) TTF_CloseFont(ui->ext->item_font);
611  if (ui->ext->descr_font) TTF_CloseFont(ui->ext->descr_font);
612  free(ui->ext);
613  }
614 
615  /* Liberation des elements */
616  if (ui->elems) {
617  /* Liberation des slots d'inventaire */
618  if (ui->elems->inventory_slots) {
619  for (int i = 0; i < ui->nbItems; i++) {
620  if (ui->elems->inventory_slots[i].texture) {
621  SDL_DestroyTexture(ui->elems->inventory_slots[i].texture);
622  }
623  }
624  free(ui->elems->inventory_slots);
625  }
626  free(ui->elems);
627  }
628 
629  /* Liberation des ecritures */
630  if (ui->ecrit) {
631  /* Liberation des chaines de caractères */
632  for (int i = 0; i < 4; i++) {
633  free(ui->ecrit->nom_txt_item[i]);
634  free(ui->ecrit->nom_txt_player[i]);
635  if (ui->ecrit->text_player[i]) freeText(ui->ecrit->text_player[i]);
636  if (ui->ecrit->text_item[i]) freeText(ui->ecrit->text_item[i]);
637  }
638 
639  /* Liberation des descriptions */
640  for (int i = 0; i < ui->ecrit->count_descr; i++) {
641  if (ui->ecrit->description[i]) freeText(ui->ecrit->description[i]);
642  }
643 
644  /* Liberation du texte de description */
645  if (ui->ecrit->text_descr) freeText(ui->ecrit->text_descr);
646  free(ui->ecrit);
647  }
648 
649  free(ui);
650 }
