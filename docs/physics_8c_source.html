10 SDL_bool checkCircleRectCollision(t_circle* circle, SDL_Rect* rect, t_collisionData* out) {
11  // Calcul du point le plus proche sur le rectangle par rapport au centre du cercl,
12  // On "contraint" la position du centre du cercle dans les limites du rectangle
13  // / C'est une projection du centre sur les bords du rectangle
14  float closestX = fmaxf(rect->x, fminf(circle->x, rect->x + rect->w));
15  float closestY = fmaxf(rect->y, fminf(circle->y, rect->y + rect->h));
16 
17  // Calcul de la distance entre le centre du cercle et le point projete
18  // C'est le vecteur qui va du point sur le rectangle vers le centre du cercle
19  float distanceX = circle->x - closestX;
20  float distanceY = circle->y - closestY;
21 
22  // On calcule la distance au carré pour éviter un appel a sqrtf qui est couteux
23  float distanceSquared = distanceX * distanceX + distanceY * distanceY;
24 
25  // Collision : si la distance au carré est inférieure au rayon au carre,
26  // cela signifie que le cercle et le rectangle se chevauchent. -> c^2 > d^2 <=> c > d
27  if (distanceSquared < (circle->radius * circle->radius)) {
28  // On calcul la vraie distance maintenant que on sais que collision
29  float distance = SDL_sqrtf(distanceSquared);
30 
31  // On calcul le vecteur normal (la direction vers laquelle le rectangle nous repousse)
32  // On inverse le vecteur distance pour pointer hors du rectangle
33  // c'est la direction dans laquelle il faut pousser le cercle pour annuler la collision
34  out->normal.x = (circle->x - closestX) / distance;
35  out->normal.y = (circle->y - closestY) / distance;
36 
37  // Calcul de la profondeur de penetration, de "combien" le cercle est rentre dans le rectangle
38  // Depth est positif, Si distance = 5 et radius = 8 -> depth = 3 (le cercle a penetre de 3)
39  // C'est la longeur qu'il faut multiplier avec la direction (donc la normal) pour sortir le cercle du rect
40  out->depth = circle->radius - distance;
41 
42  // COLLISION_LOG("Collision Cercle-Rectangle:");
43  // COLLISION_LOG(" Cercle: (%.2f,%.2f) R=%.2f", circle->x, circle->y, circle->radius);
44  // COLLISION_LOG(" Rectangle: (%d,%d) %dx%d", rect->x, rect->y, rect->w, rect->h);
45  // COLLISION_LOG(" Normal: (%.2f,%.2f) Depth=%.2f", out->normal.x, out->normal.y, out->depth);
46  return SDL_TRUE;
47  }
48  return SDL_FALSE;
49 }

53  float dy = a->y - b->y;
54  float distanceSquared = dx * dx + dy * dy; // On utilise le carré de la distance

56  float radiiSquared = radii * radii; // On utilise le carré de la somme des rayons
57

70 SDL_bool cercleInSector(SDL_FPoint target, float target_radius, SDL_FPoint origin, float currentAngle, float range, float arc) {
71  // Calcul du vecteur différence entre la cible et l'origine
72  float dx = target.x - origin.x;
73  float dy = target.y - origin.y;
74 
75  // Vérification rapide de la distance au carré (évite sqrt qui est lourd)
76  // On utilise le carré de la distance pour comparer avec le carré du rayon maximum
77  float distSq = dx * dx + dy * dy;
78 
79  // Distance maximale possible = portée du secteur + rayon de la cible
80  // (permet de détecter les cibles qui touchent le bord extérieur du secteur)
81  float maxDistSq = (range + target_radius) * (range + target_radius);
82 
83  // Si la distance au carré est supérieure à la distance maximale au carré,
84  // alors la cible est hors de portée
85  if (distSq > maxDistSq)
86  return SDL_FALSE;
87 
88  // Si la cible englobe l'origine (sa distance est inférieure à son rayon),
89  // alors elle intersecte forcément le secteur quel que soit l'angle
90  if (distSq <= target_radius * target_radius)
91  return SDL_TRUE;
92 
93  // Maintenant que les cas extremes sont traités, on calcule la vraie distance
94  float distance = sqrtf(distSq);
95 
96  // Calcul de l'angle vers la cible (en radians)
97  float targetAngle = atan2f(dy, dx);
98 
99  // Normalisation optimisée de la différence d'angle
100  // Cela garantit que angleDiff est toujours dans l'intervalle [-pi, pi]
101  // (important pour comparer correctement les angles)
102  float angleDiff = targetAngle - currentAngle;
103  if (angleDiff > M_PI) angleDiff -= 2 * M_PI;
104  if (angleDiff < -M_PI) angleDiff += 2 * M_PI;
105 
106  // Extension de l'angle autorisé pour tenir compte du rayon de la cible
107  // Plus la cible est proche, plus son rayon occupe un angle important vu depuis l'origine
108  // theta = arcsin(r/d) = angle sous lequel on voit le rayon de la cible depuis l'origine
109  float theta = asinf(fminf(target_radius / distance, 1.0f)); // Prévention contre les valeurs >1
110 
111  // L'arc effectif est la moitié de l'arc total + l'angle supplémentaire dû au rayon
112  float half_arc = arc / 2.0f + theta;
113 
114  // L'angle est-il dans l'arc d'attaque étendu?
115  // On utilise la valeur absolue car l'arc est symétrique par rapport à currentAngle
116  return fabsf(angleDiff) <= half_arc;
117 }

253  (*obstructionPoint).y = y0 * tileSize + tileSize / 2;

262

264  if (e2 >= dy) {
