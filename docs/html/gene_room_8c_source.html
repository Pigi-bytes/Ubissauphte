328 SDL_bool trouerGrille(t_grille **grille, int xdebut, int ydebut, int changementLigne, int changementColonne, t_listeBlock *lb) {
329  if (!grille) return SDL_FALSE; // Vérification de la grille
330 
331  int i = xdebut, j = ydebut;
332  if (changementLigne == 0 && changementColonne == 1) {
333  copierVal(blockByName(lb, DIRECTION_GAUCHE), &(*grille)->grille[i][j]->tiles);
334  } else if (changementLigne == 0 && changementColonne == -1) {
335  copierVal(blockByName(lb, DIRECTION_DROITE), &(*grille)->grille[i][j]->tiles);
336  } else if (changementLigne == 1 && changementColonne == 0) {
337  copierVal(blockByName(lb, DIRECTION_BAS), &(*grille)->grille[i][j]->tiles);
338  } else {
339  copierVal(blockByName(lb, DIRECTION_HAUT), &(*grille)->grille[i][j]->tiles);
340  }
341  (*grille)->grille[i][j]->val = SOL;
342 
343  // Creuser jusqu'à une case ouverte ou jusqu'à sortir des limites
344  while (i >= 0 && i < (*grille)->nbLigne && j >= 0 && j < (*grille)->nbColonne && !estOuvert((*grille)->grille[i][j], changementLigne, changementColonne)) {
345  i += changementLigne;
346  j += changementColonne;
347  if (i >= 0 && i < (*grille)->nbLigne && j >= 0 && j < (*grille)->nbColonne) {
348  (*grille)->grille[i][j]->val = SORTIE;
349  }
350  }
351 
352  if (i < 0 || i >= (*grille)->nbLigne || j < 0 || j >= (*grille)->nbColonne) {
353  for (i = xdebut, j = ydebut; i >= 0 && i < (*grille)->nbLigne && j >= 0 && j < (*grille)->nbColonne; i += changementLigne, j += changementColonne) {
354  (*grille)->grille[i][j]->val = OBSTACLE;
355  }
356  return SDL_FALSE;
357  }
358  if ((*grille)->grille[i + changementLigne][j + changementColonne]->val == OBSTACLE) {
359  (*grille)->grille[i + changementLigne][j + changementColonne]->val = SORTIE;
360  }
361 
362  return SDL_TRUE;
